#!/bin/bash
# Secure File Organizer v1.2

TARGET_DIR="/var/log/analysis"
ARCHIVE_DIR="/mnt/secure_archive"

echo "[*] Initiating secure sort..."
find "$TARGET_DIR" -type f -name "*.pcap" \
     -exec mv {} "$ARCHIVE_DIR/pcaps/" \; \
     -exec chmod 600 "$ARCHIVE_DIR/pcaps/"{} \;

echo "[+] Operation completed securely."
